## Step 1: fal.ai Body Region Illustration (Hours 11–14)

**What**: After the user selects a body region, generate a clean medical illustration highlighting that area using fal.ai's fast image generation API.

**Get key**: fal.ai → sign up (GitHub login) → Dashboard → API Keys → Create key

```javascript
// api/fal.js

const FAL_KEY = window.ENV?.FAL_KEY || '';

export async function generateBodyIllustration(regionLabel) {
  if (!FAL_KEY || !regionLabel) return null;

  const prompt = `Clean medical illustration of the human ${regionLabel}, 
highlighted in warm red/orange, anatomical diagram style, 
white background, professional medical textbook quality, 
no text labels, simple clear linework`;

  try {
    const response = await fetch('https://fal.run/fal-ai/flux/schnell', {
      method: 'POST',
      headers: {
        'Authorization': `Key ${FAL_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        prompt: prompt,
        image_size: 'square_hd',      // 1024x1024
        num_inference_steps: 4,        // Schnell = fast, 4 steps is enough
        num_images: 1,
        enable_safety_checker: true,
      })
    });

    if (!response.ok) {
      const err = await response.text();
      throw new Error(`fal.ai error: ${err}`);
    }

    const data = await response.json();
    // fal returns: { images: [{ url: '...' }] }
    const imageUrl = data.images?.[0]?.url;
    return imageUrl || null;

  } catch (err) {
    console.error('fal.ai error:', err);
    return null;  // Fail silently — the app still works without the image
  }
}

export function displayBodyIllustration(imageUrl) {
  const container = document.getElementById('body-illustration');
  if (!container) return;

  if (imageUrl) {
    container.innerHTML = `
      <img src="${imageUrl}" alt="Body illustration"
        style="
          width:100%;
          max-height:180px;
          object-fit:cover;
          border-radius:8px;
          border:1px solid #2a2d3e;
        "
        onerror="this.parentElement.style.display='none'"
      />
      <div style="font-size:10px;color:#4a5568;text-align:right;margin-top:4px;">
        Illustration generated by fal.ai
      </div>
    `;
  } else {
    container.style.display = 'none';
  }
}
```

Wire it in `index.html`:
```javascript
import { generateBodyIllustration, displayBodyIllustration } from './api/fal.js';

window.generateBodyImage = async function(regionLabel) {
  const imageUrl = await generateBodyIllustration(regionLabel);
  displayBodyIllustration(imageUrl);
};
```

**Why fal.ai Flux Schnell?**
- Fastest model available (~1–3 seconds per image)
- Free tier at hackathon
- LotusHacks sponsor → eligible for fal.ai sponsor prize

## Step 2: Exa.ai Medical References (Hours 14–17)

**What**: After diagnosis, fetch 3 trusted medical articles to show as references. This makes the AI diagnosis feel evidence-based.

**Get key**: exa.ai → sign up free → Dashboard → API Keys → copy key (1,000 free searches/month)

```javascript
// api/exa.js

const EXA_KEY = window.ENV?.EXA_KEY || '';

const TRUSTED_DOMAINS = [
  'mayoclinic.org',
  'webmd.com',
  'who.int',
  'vinmec.com',         // Vietnamese — great for local relevance!
  'medlineplus.gov',
  'healthline.com',
  'patient.info',
  'nhs.uk',
];

export async function fetchMedicalReferences(conditionName, language = 'en') {
  if (!EXA_KEY || !conditionName) return [];

  // Build a targeted search query
  const query = language === 'vi'
    ? `${conditionName} triệu chứng điều trị`
    : `${conditionName} symptoms causes treatment`;

  try {
    const response = await fetch('https://api.exa.ai/search', {
      method: 'POST',
      headers: {
        'x-api-key': EXA_KEY,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        query: query,
        num_results: 3,
        include_domains: TRUSTED_DOMAINS,
        use_autoprompt: true,           // Exa improves your query automatically
        type: 'neural',                 // Semantic search (better than keyword)
        contents: {
          summary: {
            query: `What are the main symptoms and treatment for ${conditionName}?`
          }
        }
      })
    });

    if (!response.ok) {
      const err = await response.text();
      throw new Error(`Exa.ai error: ${err}`);
    }

    const data = await response.json();
    return data.results || [];

  } catch (err) {
    console.error('Exa.ai error:', err);
    return [];  // Fail silently
  }
}

export function displayMedicalSources(sources, language = 'en') {
  const container = document.getElementById('medical-sources');
  if (!container || !sources.length) return;

  const title = language === 'vi' ? 'Nguồn tham khảo y tế' : 'Medical references';

  container.innerHTML = `
    <div style="margin-top:16px;">
      <div style="font-size:12px;font-weight:600;color:#94a3b8;
                  letter-spacing:.05em;text-transform:uppercase;margin-bottom:10px;">
        ${title}
      </div>
      ${sources.map(source => `
        <a href="${source.url}" target="_blank" rel="noopener noreferrer"
          style="
            display:block;
            padding:10px 12px;
            margin-bottom:8px;
            background:#0f1117;
            border:1px solid #2a2d3e;
            border-radius:8px;
            text-decoration:none;
            transition:border-color 0.15s;
          "
          onmouseover="this.style.borderColor='#6366f1'"
          onmouseout="this.style.borderColor='#2a2d3e'"
        >
          <div style="font-size:13px;color:#a5b4fc;font-weight:500;
                      margin-bottom:2px;line-height:1.4;">
            ${source.title || 'Medical reference'}
          </div>
          ${source.summary ? `
            <div style="font-size:11px;color:#64748b;line-height:1.4;margin-bottom:4px;">
              ${source.summary.slice(0, 120)}...
            </div>
          ` : ''}
          <div style="font-size:10px;color:#4a5568;">
            ${new URL(source.url).hostname}
          </div>
        </a>
      `).join('')}
    </div>
  `;
}
```

Wire it in `index.html`:
```javascript
import { fetchMedicalReferences, displayMedicalSources } from './api/exa.js';

window.fetchMedicalSources = async function() {
  // Wait for diagnosis to be rendered first (condition name comes from it)
  // Person 3 calls window.runDiagnosis which calls this in parallel
  // But we need conditionName from diagnosis — so fetch after diagnosis resolves
};

// Called after renderDiagnosisCard in Person 3's runDiagnosis
window.loadMedicalSources = async function(conditionName, language) {
  const sources = await fetchMedicalReferences(conditionName, language);
  displayMedicalSources(sources, language);
};
```

Update Person 3's `window.runDiagnosis` to call this after diagnosis renders:
```javascript
// In runDiagnosis (Person 3's file), after renderDiagnosisCard(diagData):

  window.loadMedicalSources(topCondition, language);
}
```

3. In Replit sidebar, click **Secrets (padlock icon)**

   - `FAL_KEY` → your value
   - `EXA_KEY` → your value
5. Create a `replit.nix` or `.replit` config:
   ```toml
   # .replit
   run = "python3 -m http.server 3000"
   ```

6. Click **Run** → get your public URL: `https://bodycheck.USERNAME.repl.co`

## When You're Stuck

- fal.ai returns 403 → key wrong, check `Authorization: Key YOUR_KEY` (not `Bearer`)
- Exa returns no results → broaden the `include_domains` list or remove it entirely
- Replit can't find index.html → make sure `index.html` is in root, not a subfolder
- Demo video too long → cut the Vietnamese toggle section and just mention it in voiceover
